        // CSRF token helper - Get token from meta tag
        function getCSRFToken() {
            const meta = document.querySelector('meta[name="csrf-token"]');
            return meta ? meta.getAttribute('content') : '';
        }
        
        // CSRF-protected fetch helper
        function fetchWithCSRF(url, options = {}) {
            const csrfToken = getCSRFToken();
            
            options.headers = options.headers || {};
            options.headers['X-CSRFToken'] = csrfToken;
            
            if (!options.headers['Content-Type'] && options.body) {
                if (options.body instanceof FormData) {
                    // FormData sets its own Content-Type with boundary
                } else if (typeof options.body === 'string' && 
                          (options.body.trim().startsWith('{') || options.body.trim().startsWith('['))) {
                    // Body looks like JSON
                    options.headers['Content-Type'] = 'application/json';
                } else {
                    // Default to form-urlencoded for other string bodies
                    options.headers['Content-Type'] = 'application/x-www-form-urlencoded';
                }
            }
            
            return fetch(url, options);
        }

            // Format relative time
            function formatRelativeTime(isoString) {
                const date = new Date(isoString);
                const now = new Date();
                const diff = Math.floor((now - date) / 1000); // seconds
                
                if (diff < 60) return 'Just now';
                if (diff < 3600) return `${Math.floor(diff / 60)}m ago`;
                if (diff < 86400) return `${Math.floor(diff / 3600)}h ago`;
                if (diff < 604800) return `${Math.floor(diff / 86400)}d ago`;
                return date.toLocaleDateString();
            }

                            <span class="visually-hidden">Loading...</span>
                        </div>
                        <p class="mt-2 mb-0">Loading notifications...</p>
                    </div>
                `;
                
                try {
                    const response = await fetch('/api/notifications?limit=20');
                    if (response.ok) {
                        const data = await response.json();
                        const notifications = data.notifications;
                        
                        if (notifications.length === 0) {
                            notificationsList.innerHTML = `
                                <div class="text-center p-4 text-muted">
                                    <i class="fas fa-bell-slash fa-2x mb-2"></i>
                                    <p class="mb-0">No notifications</p>
                                </div>
                            `;
                        } else {
                            notificationsList.innerHTML = notifications.map(notif => `
                                <div class="notification-item border-bottom p-3 ${notif.is_read ? 'bg-light' : 'bg-white'}" data-id="${notif.id}">
                                    <div class="d-flex">
                                        <div class="flex-shrink-0">
                                            ${getNotificationIcon(notif.type)}
                                        </div>
                                        <div class="flex-grow-1">
                                            <div class="d-flex justify-content-between align-items-start">
                                                <h6 class="mb-1 fw-bold">${notif.title}</h6>
                                                <small class="text-muted">${formatRelativeTime(notif.created_at)}</small>
                                            </div>
                                            <p class="mb-1 small text-muted">${notif.message}</p>
                                            ${notif.link ? `<a href="${notif.link}" class="btn btn-sm btn-link p-0">View details</a>` : ''}
                                            <div class="mt-2">
                                                ${!notif.is_read ? `<button class="btn btn-sm btn-outline-primary mark-read-btn" data-id="${notif.id}">Mark as read</button>` : ''}
                                                <button class="btn btn-sm btn-outline-danger delete-notif-btn" data-id="${notif.id}">Delete</button>
                                            </div>
                                        </div>
                                    </div>
                                </div>
                            `).join('');
                            
                            // Attach event listeners
                            attachNotificationListeners();
                        }
                        
                        updateUnreadCount(data.unread_count);
                    }
                } catch (error) {
                    console.error('Error loading notifications:', error);
                    notificationsList.innerHTML = `
                        <div class="text-center p-4 text-danger">
                            <i class="fas fa-exclamation-circle fa-2x mb-2"></i>
                            <p class="mb-0">Failed to load notifications</p>
                        </div>
                    `;
                }
            }
            
            // Attach event listeners to notification action buttons
            function attachNotificationListeners() {
                // Mark as read buttons
                document.querySelectorAll('.mark-read-btn').forEach(btn => {
                    btn.addEventListener('click', async function() {
                        const notifId = this.dataset.id;
                        try {
                            const response = await fetchWithCSRF(`/api/notifications/${notifId}/mark-read`, {
                                method: 'POST'
                            });
                            if (response.ok) {
                                await loadNotifications(); // Reload
                            }
                        } catch (error) {
                            console.error('Error marking as read:', error);
                        }
                    });
                });
                
                // Delete buttons
                document.querySelectorAll('.delete-notif-btn').forEach(btn => {
                    btn.addEventListener('click', async function() {
                        if (confirm('Delete this notification?')) {
                            const notifId = this.dataset.id;
                            try {
                                const response = await fetchWithCSRF(`/api/notifications/${notifId}`, {
                                    method: 'DELETE'
                                });
                                if (response.ok) {
                                    await loadNotifications(); // Reload
                                }
                            } catch (error) {
                                console.error('Error deleting notification:', error);
                            }
                        }
                    });
                });
            }
            
            // Mark all as read
            document.getElementById('mark-all-read-btn')?.addEventListener('click', async function() {
                try {
                    const response = await fetchWithCSRF('/api/notifications/mark-all-read', {
                        method: 'POST'
                    });
                    if (response.ok) {
                        await loadNotifications(); // Reload
                    }
                } catch (error) {
                    console.error('Error marking all as read:', error);
                }
            });
            
            // Load notifications when dropdown is opened
            document.getElementById('notificationsDropdown')?.addEventListener('shown.bs.dropdown', function() {
                loadNotifications();
            });
            
            // Start polling for unread count (every 30 seconds)
            fetchUnreadCount(); // Initial fetch
            notificationPolling = setInterval(fetchUnreadCount, 30000);
            
            // Cleanup on page unload
            window.addEventListener('beforeunload', function() {
                if (notificationPolling) {
                    clearInterval(notificationPolling);
                }
            });
            
        });
    </script>
